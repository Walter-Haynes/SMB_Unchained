14 inline Pixel AddBlend( Pixel a_Color1, Pixel a_Color2 )
15 {
16  const unsigned int r = (a_Color1 & REDMASK) + (a_Color2 & REDMASK);
17  const unsigned int g = (a_Color1 & GREENMASK) + (a_Color2 & GREENMASK);
18  const unsigned int b = (a_Color1 & BLUEMASK) + (a_Color2 & BLUEMASK);
19  const unsigned r1 = (r & REDMASK) | (REDMASK * (r >> 24));
20  const unsigned g1 = (g & GREENMASK) | (GREENMASK * (g >> 16));
21  const unsigned b1 = (b & BLUEMASK) | (BLUEMASK * (b >> 8));
22  return (r1 + g1 + b1);
23 }

25 // subtractive blending
26 inline Pixel SubBlend( Pixel a_Color1, Pixel a_Color2 )
27 {
28  int red = (a_Color1 & REDMASK) - (a_Color2 & REDMASK);
29  int green = (a_Color1 & GREENMASK) - (a_Color2 & GREENMASK);
30  int blue = (a_Color1 & BLUEMASK) - (a_Color2 & BLUEMASK);
31  if (red < 0) red = 0;
32  if (green < 0) green = 0;
33  if (blue < 0) blue = 0;
34  return static_cast<Pixel>(red + green + blue);
35 }
